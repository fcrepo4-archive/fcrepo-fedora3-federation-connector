017package org.fcrepo.connector.fedora3;
018
019import java.util.List;
020
021/**
022 * An interface to expose enough information about a Fedora 3 datastream to
023 * import it into fedora 4.  All metadata is reprsented here, even those fields
024 * that are meaningless in the Fedora 4 architecture.
025 * 
026 * @author Michael Durbin
027 */
028public interface FedoraDatastreamRecord {
029
030    /**
031     * Gets the pid of the object whose datastream is described by this record.
032     */
033    public String getPid();
034
035    /**
036     * Gets the DSID for the datastream described by this record.
037     */
038    public String getId();
039
040    /**
041     * Get the control group for the datastream described by this record.  This
042     * value, while significant in Fedora 3, is only present as an indicator of
043     * historic status, since storage management is handled through another
044     * mechanism in Fedora 4.
045     */
046    public String getControlGroup();
047
048    /**
049     * Gets the state for the datastream described by this record.
050     */
051    public String getState();
052
053    /**
054     * Gets the 'versionable' value for the datastream described by this
055     * record.  This value does not imply anything about whether or how many
056     * versions of this datastream exists (there is always at least one) but
057     * instead was an indicator of whether an update to the described
058     * datastream in Fedora 3 would overwrite the current version or create a
059     * new one.  In the context of fedora 4, this value is just to preserve
060     * the historical record.
061     */
062    public boolean getVersionable();
063
064    /**
065     * Returns a non-null, non-empty list of FedoraDatastreamVersionRecord
066     * objects representing the ordered history of this datastream from most
067     * recent to oldest.
068     */
069    public List<FedoraDatastreamVersionRecord> getHistory();
070
071    /**
072     * Gets the most recent version of the datastream.  This method is just a
073     * shortcut for getHistory().get(0).
074     */
075    public FedoraDatastreamVersionRecord getCurrentVersion();
076
077}
