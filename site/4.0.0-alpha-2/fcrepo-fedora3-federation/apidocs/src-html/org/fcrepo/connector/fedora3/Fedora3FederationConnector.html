113    public void initialize(NamespaceRegistry registry,
114            NodeTypeManager nodeTypeManager)
115        throws RepositoryException, IOException {
116        super.initialize(registry, nodeTypeManager);
117
118        LOGGER.trace("Initializing");
119
120        if (nodeTypeManager != null) {  // only null for unit tests
121            nodeTypeManager.registerNodeTypes(
122                    getClass().getClassLoader()
123                            .getResourceAsStream("fedora3-node-types.cnd"),
124                    true);
125            LOGGER.debug("Loaded node types from {}.",
126                    getClass().getClassLoader()
127                            .getResource("fedora3-node-types.cnd"));
128        }
129
130        try {
131            if (fedoraUrl != null && username != null && password != null) {
132                f3 = new RESTFedora3DataImpl(fedoraUrl, username,
133                        password);
134            } else {
135                throw new RepositoryException("Requred parameters missing, "
136                        + "ensure that \"fedoraUrl\", \"username\" and "
137                        + " \"password\" are set!");
138            }
139        } catch (Throwable t) {
140            throw new RepositoryException("Error starting fedora connector!",
141                    t);
142        }
143        organizer = new FlatTruncatedOrganizer(f3);
144        LOGGER.trace("Initialized");
145    }

147    /**
148     * {@inheritDoc}
149     */
150    public Document getDocumentById(String idStr) {
151        LOGGER.trace("getDocumentById {}", idStr);
152        ID id = new ID(idStr);
153        DocumentWriter writer = newDocument(idStr);
154        writer.setNotQueryable();
155        if (id.isRootID()) {
156            // return a root object
157            writer.setPrimaryType(JcrConstants.NT_FOLDER);
158            writer.addMixinType(NT_F3_REPOSITORY);
159            addRepositoryChildren(writer, idStr);
160            return writer.document();
161        } else if (organizer.isOrganizationalNode(idStr) && !id.isRootID()) {
162            writer.setPrimaryType(JcrConstants.NT_FOLDER);
163            writer.addMixinType(NT_F3_GROUP);
164            for (String childId : organizer.getChildrenForId(idStr)) {
165                if (organizer.isOrganizationalNode(childId)) {
166                    writer.addChild(childId, childId);
167                } else {
168                    writer.addChild(ID.objectID(childId).getId(),
169                            ID.objectID(childId).getName());
170                }
171            }
172            return writer.document();
173        } else if (id.isObjectID()) {
174            // return an object node
175            FedoraObjectRecord o = f3.getObjectByPid(id.getPid());
176            writer.setPrimaryType(JcrConstants.NT_FOLDER);
177            writer.setParent(ID.ROOT_ID.getId());
178            addObjectProperties(writer, o);
179            addObjectChildren(writer, o);
180            return writer.document();
181        } else if (id.isDatastreamID()) {
182            // return a datastream node
183            writer.setPrimaryType(JcrConstants.NT_FILE);
184            writer.setParent(id.getParentId());
185            FedoraDatastreamRecord ds
186                = f3.getDatastream(id.getPid(), id.getDSID());
187            addDatastreamProperties(writer, ds);
188            ID contentId = ID.contentID(id.getPid(), id.getDSID());
189            writer.addChild(contentId.getId(), contentId.getName());
190            return writer.document();
191        } else if (id.isContentID()) {
192            // return a content node
193            FedoraDatastreamRecord ds = f3.getDatastream(id.getPid(),
194                    id.getDSID());
195            writer.setPrimaryType(JcrConstants.NT_RESOURCE);
196            writer.setParent(id.getParentId());
197            addDatastreamContentProperties(writer, ds);
198            return writer.document();
199        } else {
200            return null;
201        }
202    }
203
204    private void addRepositoryChildren(DocumentWriter writer, String idStr) {
205        ID id = new ID(idStr);
206        List<String> childPids = organizer.getChildrenForId(idStr);
207        for (String childPid : childPids) {
208            ID childId = ID.objectID(childPid);
209            LOGGER.trace("Added child " + childId.getId());
210            writer.addChild(childId.getId(), childId.getName());
211        }
212    }

214    /**
215     * Adds the Fedora 3 object mixin type and Fedora 3 properties for the
216     * given obejct.  This method also adds any JCR base or Fedora 4 properties
217     * to which the existing Fedora 3 properties map.
218     *
219     * This method doesn't expose versioning from fedora 3 and instead only
220     * includes information from the latest version.
221     */
222    private void addObjectProperties(DocumentWriter writer,
223            FedoraObjectRecord o) {
224        // Fedora 3 Properties
225        writer.addMixinType(NT_F3_OBJECT);
226        writer.addProperty(F3_PID, o.getPid());
227        addOptionalProperty(writer, F3_OBJ_STATE, o.getState());
228        addOptionalProperty(writer, F3_OBJ_LABEL, o.getLabel());
229        addOptionalProperty(writer, F3_OBJ_OWNER_ID, o.getOwnerIds());
230        addOptionalProperty(writer, F3_OBJ_CREATED_DATE, o.getCreatedDate());
231        addOptionalProperty(writer, F3_OBJ_LAST_MODIFIED_DATE,
232                o.getModificationDate());
233
234        // JCR Properties
235        addOptionalProperty(writer, JCR_CREATED, o.getCreatedDate());
236        addOptionalProperty(writer, JCR_LASTMODIFIED, o.getModificationDate());
237    }
238
239    /**
240     * Adds the Fedora 3 datastream mixin type and Fedora 3 datastream
241     * properties for the given datastream.  This method also adds the fedora 4
242     * "datastream" mixin and relevant properties.
243     *
244     * This method does not include version information and only exposes the
245     * latest version of a datastream.
246     */
247    private void addDatastreamProperties(DocumentWriter writer,
248            FedoraDatastreamRecord ds) {
249        // Fedora 3 Datastream Properties
250        writer.addMixinType(NT_F3_DATASTREAM);
251        writer.addProperty(F3_DSID, ds.getId());
252        writer.addProperty(F3_DS_CONTROL_GROUP, ds.getControlGroup());
253        writer.addProperty(F3_DS_STATE, ds.getState());
254        writer.addProperty(F3_DS_VERSIONABLE, ds.getVersionable());
255        FedoraDatastreamVersionRecord dsVer = ds.getCurrentVersion();
256        writer.addProperty(F3_DS_VERSION_ID, dsVer.getVersionId());
257        addOptionalProperty(writer, F3_DS_CREATED, dsVer.getCreatedDate());
258        addOptionalProperty(writer, F3_DS_LABEL, dsVer.getLabel());
259        writer.addProperty(F3_DS_MIME_TYPE, dsVer.getMimeType());
260        addOptionalProperty(writer, F3_DS_FORMAT_URI, dsVer.getFormatURI());
261        addOptionalProperty(writer, F3_DS_ALT_IDS, dsVer.getAltIDs());
262        writer.addProperty(F3_DS_SIZE, dsVer.getContentLength());
263        addOptionalProperty(writer, F3_DS_CONTENT_DIGEST_TYPE,
264                dsVer.getContentDigestType());
265        addOptionalProperty(writer, F3_DS_CONTENT_DIGEST,
266                dsVer.getContentDigest());
267
268        // Fedora 4 Datastream Properties
269        writer.addMixinType(FEDORA_DATASTREAM);
270        if (getContext() != null) { //only null for unit tests
271            DateTimeFactory f = factories().getDateFactory();
272            writer.addProperty(JCR_LASTMODIFIED,
273                    f.create(dsVer.getCreatedDate()));
274            FedoraDatastreamVersionRecord firstVer
275                = ds.getHistory().get(ds.getHistory().size() - 1);
276            writer.addProperty(JCR_CREATED,
277                        f.create(firstVer.getCreatedDate()));
278        }
279    }
280
281    private void addDatastreamContentProperties(DocumentWriter writer,
282            FedoraDatastreamRecord ds) {
283        writer.addMixinType(FEDORA_BINARY);
284        try {
285            BinaryValue binary = new Fedora3DatastreamBinaryValue(ds);
286            writer.addProperty(JcrConstants.JCR_DATA, binary);
287            LOGGER.trace("{} size: {}", ds.getId(), binary.getSize());
288            LOGGER.trace("{} hash: {}", ds.getId(), binary.getHexHash());
289            writer.addProperty(CONTENT_DIGEST, ContentDigest.
290                    asURI("SHA-1", binary.getHexHash()));
291            writer.addProperty(CONTENT_SIZE, binary.getSize());
292        } catch (Exception ex) {
293            throw new RuntimeException(ex);
294        }
295        writer.addProperty(JcrConstants.JCR_MIME_TYPE,
296                ds.getCurrentVersion().getMimeType());
297
298    }

300    /**
301     * A helper method that adds a property (or list of properties) if the
302     * value isn't null.  This method also conveniently converts Date values to
303     * DateTime values.
304     */
305    private void addOptionalProperty(DocumentWriter writer, String name,
306            Object value) {
307        if (value != null) {
308            if (value instanceof List) {
309                writer.addProperty(name, ((List) value).toArray());
310            } else if (value instanceof Date) {
311                if (getContext() != null) { //only null for unit tests
312                    DateTimeFactory f = factories().getDateFactory();
313                    writer.addProperty(name, f.create((Date) value));
314                }
315            } else {
316                writer.addProperty(name, value);
317            }
318        }
319    }

322    private void addObjectChildren(DocumentWriter writer,
323            FedoraObjectRecord object) {
324        for (String dsidStr : object.listDatastreamIds()) {
325            ID dsid = ID.datastreamID(object.getPid(), dsidStr);
326            writer.addChild(dsid.getId(), dsid.getName());
327        }
328    }

334        LOGGER.info("getDocumentId {}", externalPath);
335        return getIdFromPath(externalPath);
336    }

338    /**
339     * The root path is always '/', organizational node paths are made up up
340     * any number of organizational node ids concatenated together, while all
341     * other nodes are made up of any number of organizational node paths
342     * concatenated together followed by an object, datastream or content ID.
343     *
344     * This method uses that to determine the id of the node at the given path.
345     */
346    protected String getIdFromPath(String path) {
347        int nextBreak = path.indexOf('/', 1);
348        if (nextBreak == -1) {
349            return path;
350        } else {
351            String nextChunk = path.substring(0, nextBreak);
352            if (organizer.isOrganizationalNode(nextChunk)) {
353                return getIdFromPath(path.substring(nextBreak));
354            } else {
355                return path;
356            }
357        }
358    }

360    @Override
361    public Collection<String> getDocumentPathsById(String id) {
362        LOGGER.info("getDocumentPathsById {}", id);
363        return Collections.singletonList(buildPath(id, id));
364    }

366    /**
367     * The root path is always '/', organizational node paths are made up up
368     * any number of organizational node ids concatenated together, while all
369     * other nodes are made up of any number of organizational node paths
370     * concatenated together followed by an object, datastream or content ID.
371     *
372     * This method uses that to build a path from an ID.
373     */
374    protected String buildPath(String path, String currentId) {
375        String parentId = organizer.getParentForId(currentId);
376        if (parentId == null) {
377            return path;
378        } else {
379            return buildPath(parentId + path, parentId);
380        }
381    }

385    /**
386     * Checks if a document with the given id exists.
387     * @param idStr a {@code non-null} string representing the identifier
388     *              within the system whose existence is being queried in this
389     *              federation.
390     */
391    public boolean hasDocument(String idStr) {
392        LOGGER.info("hasDocument {}", idStr);
393        if (organizer.isOrganizationalNode(idStr)) {
394            return true;
395        }
396        ID id = new ID(idStr);
397        return (id.isRootID()
398                || (id.isObjectID() && f3.doesObjectExist(id.getPid())
399                || ((id.isDatastreamID() || id.isContentID())
400                && f3.doesDatastreamExist(id.getPid(), id.getDSID()))));
401    }

403    /**
404     * Gets an ExternalBinaryValue that exposes access to content nodes in
405     * the federation.
406     */
407    public ExternalBinaryValue getBinaryValue(String idStr) {
408        ID id = new ID(idStr);
409        FedoraDatastreamRecord ds = f3.getDatastream(id.getPid(),
410                id.getDSID());
411        try {
412            return new Fedora3DatastreamBinaryValue(ds);
413        } catch (Exception e) {
414            throw new RuntimeException(e);
415        }
416    }
417
418    public class Fedora3DatastreamBinaryValue extends ExternalBinaryValue {
419
420        private static final long serialVersionUID = 1L;
421
422        private FedoraDatastreamRecord ds;
423
424        Fedora3DatastreamBinaryValue(FedoraDatastreamRecord ds)
425            throws Exception {
426            super(new BinaryKey(ds.getCurrentVersion().getSha1()),
427                    Fedora3FederationConnector.this.getSourceName(),
428                    ID.contentID(ds.getPid(), ds.getId()).getId(),
429                    ds.getCurrentVersion().getContentLength(), null, null);
430            this.ds = ds;
431        }
432
433        /**
434         * Gets the InputStream for the content.
435         */
436        public InputStream getStream() throws RepositoryException {
437            try {
438                return ds.getCurrentVersion().getStream();
439            } catch (Exception e) {
440                throw new RepositoryException(e);
441            }
442        }
443
444        /**
445         * Overrides the superclass to return the MIME type declared on the
446         * fedora 3 datastream whose content is exposed by this BinaryValue.
447         */
448        public String getMimeType() {
449            return ds.getCurrentVersion().getMimeType();
450        }
451    }
