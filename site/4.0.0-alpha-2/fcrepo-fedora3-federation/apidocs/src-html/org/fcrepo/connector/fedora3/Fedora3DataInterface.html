017package org.fcrepo.connector.fedora3;
018
019import java.util.List;
020
021/**
022 * An base class that encapsulates the logic to access content from a fedora 3
023 * repository.  This abstract class, while exposing the methods neccessary to
024 * implement a federation over fedora 3 content is agnostic about the way data
025 * is retrieved from said repository
026 * 
027 * @author Michael Durbin
028 */
029public interface Fedora3DataInterface {
030
031    /**
032     * Gets a FedoraObjectRecord that encapsulates a summary of the object with
033     * the given pid in the fedora 3 repository exposed through this interface.
034     */
035    public FedoraObjectRecord getObjectByPid(String pid);
036
037    /**
038     * Determines if an object with the given pid exists in the fedora 3
039     * repository exposed through this interface.
040     */
041    public boolean doesObjectExist(String pid);
042
043    /**
044     * Gets a page of object pids that exist in the repository.
045     */
046    public List<String> getObjectPids(int offset, int pageSize);
047
048    /**
049     * Gets the size (total number of objects) of the underlying fedora 3
050     * repository.
051     */
052    public long getSize();
053
054    /**
055     * Gets information about a given datastream for a given pid.
056     */
057    public FedoraDatastreamRecord getDatastream(String pid,
058            String dsid);
059
060    /**
061     * Determines if an object with the given pid exists and has a datastream
062     * with the given dsid.
063     */
064    public boolean doesDatastreamExist(String pid, String dsid);
065}
