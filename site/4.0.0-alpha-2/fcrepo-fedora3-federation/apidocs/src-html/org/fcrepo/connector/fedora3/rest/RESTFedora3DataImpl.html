017package org.fcrepo.connector.fedora3.rest;
018
019import com.yourmediashelf.fedora.client.FedoraClient;
020import com.yourmediashelf.fedora.client.FedoraClientException;
021import com.yourmediashelf.fedora.client.FedoraCredentials;
022import com.yourmediashelf.fedora.generated.access.FedoraRepository;
023import org.fcrepo.connector.fedora3.Fedora3DataInterface;
024import org.fcrepo.connector.fedora3.FedoraDatastreamRecord;
025import org.fcrepo.connector.fedora3.FedoraObjectRecord;
026import org.slf4j.Logger;
027
028import java.io.BufferedReader;
029import java.io.IOException;
030import java.io.InputStreamReader;
031import java.net.MalformedURLException;
032import java.util.ArrayList;
033import java.util.List;
034
035import static org.slf4j.LoggerFactory.getLogger;
036
037/**
038 * An implementation of Fedora3DataInterface that uses the REST API to access
039 * fedora content.
040 * 
041 * @author Michael Durbin
042 */
043public class RESTFedora3DataImpl implements Fedora3DataInterface {
044
045    private static final Logger LOGGER
046        = getLogger(RESTFedora3DataImpl.class);
047
048    private FedoraClient fc;
049
050    /**
051     * The cached size (number of objects) of the repository.  This allows
052     * subsequent calls to getSize() to be cheap.
053     */
054    private long size = -1L;
055
056    /**
057     * Constructor with credentials necessary for a connection to fedora's REST
058     * API.
059     */
060    public RESTFedora3DataImpl(String fedoraUrl, String username,
061            String password) throws MalformedURLException,
062            FedoraClientException {
063        initialize(new FedoraClient(
064                new FedoraCredentials(fedoraUrl, username, password)));
065    }
066
067    private void initialize(FedoraClient fc) throws FedoraClientException {
068        this.fc = fc;
069        FedoraRepository r = FedoraClient.describeRepository().execute(fc)
070                .getRepositoryInfo();
071        LOGGER.debug("Initialized connection to fedora "
072                + r.getRepositoryVersion() + " at "
073                + r.getRepositoryBaseURL() + " with the resource index enabled"
074                + " and " + getSize() + " objects.");
075    }
076
077
078    /**
079     * {@inheritDoc}
080     */
081    public FedoraObjectRecord getObjectByPid(String pid) {
082        try {
083            return new ObjectProfileObjectRecordImpl(
084                    FedoraClient.getObjectProfile(pid)
085                    .execute(fc).getObjectProfile(),
086                    FedoraClient.listDatastreams(pid)
087                    .execute(fc).getDatastreams());
088        } catch (FedoraClientException e) {
089            throw new RuntimeException(e);
090        }
091    }
092
093    /**
094     * {@inheritDoc}
095     */
096    public boolean doesObjectExist(String pid) {
097        try {
098            return FedoraClient.getObjectProfile(pid).execute(fc).getStatus()
099                    == 200;
100        } catch (FedoraClientException ex) {
101            throw new RuntimeException(ex);
102        }
103    }
104
105    /**
106     * {@inheritDoc}
107     *
108     * The current implementation relies on the following ITQL query against
109     * the resource index to fetch results.
110     *
111     * <pre>
112     *  {@code
113     *    select     $object
114     *    from       <#ri>
115     *    where      $object <info:fedora/fedora-system:def/model#hasModel>
116     *        <info:fedora/fedora-system:FedoraObject-3.0>
117     *    order by   $object
118     *    limit      -pageSize-
119     *    offset     -offset-
120     *  }
121     * </pre>
122     */
123    public List<String> getObjectPids(int offset, int pageSize) {
124        String query = "select $object"
125                + " from <#ri>"
126                + " where $object"
127                + " <info:fedora/fedora-system:def/model#hasModel>"
128                + " <info:fedora/fedora-system:FedoraObject-3.0>"
129                + " order by $object"
130                + " limit " + pageSize
131                + " offset " + offset;
132        try {
133            ArrayList<String> pids = new ArrayList<String>();
134            BufferedReader r = new BufferedReader(
135                    new InputStreamReader(
136                            FedoraClient.riSearch(query).lang("itql")
137                                    .format("csv").execute(fc)
138                                    .getEntityInputStream()));
139            r.readLine().equals("\"object\"");
140            String objectUri = null;
141            while ((objectUri = r.readLine()) != null) {
142                pids.add(objectUri.substring("info:fedora/".length()));
143            }
144            return pids;
145        } catch (FedoraClientException e) {
146            throw new RuntimeException(e);
147        } catch (IOException e) {
148            throw new RuntimeException(e);
149        }
150    }
151
152    /**
153     * {@inheritDoc}
154     *
155     * The current implementation relies on the following ITQL query against
156     * the resource index to determine repository size.
157     *
158     * <pre>
159     *  {@code
160     *    select count(
161     *        select     $object
162     *        from       <#ri>
163     *        where      $object <info:fedora/fedora-system:def/model#hasModel>
164     *            <info:fedora/fedora-system:FedoraObject-3.0>)
165     *    from <#ri>
166     *    where $a $b $c
167     *  }
168     * </pre>
169     */
170    public long getSize() {
171        if (size == -1) {
172            String query = "select count("
173                    + " select $object"
174                    + " from <#ri>"
175                    + " where $object"
176                    + " <info:fedora/fedora-system:def/model#hasModel>"
177                    + " <info:fedora/fedora-system:FedoraObject-3.0>"
178                    + ") from <#ri> where $a $b $c";
179            try {
180                BufferedReader r = new BufferedReader(
181                        new InputStreamReader(
182                                FedoraClient.riSearch(query).lang("itql")
183                                        .format("csv").execute(fc)
184                                        .getEntityInputStream()));
185                r.readLine().equals("\"k0\"");
186                size =  Long.parseLong(r.readLine());
187            } catch (FedoraClientException e) {
188                throw new RuntimeException(e);
189            } catch (IOException e) {
190                throw new RuntimeException(e);
191            }
192        }
193        return size;
194    }
195
196    /**
197     * {@inheritDoc}
198     */
199    public FedoraDatastreamRecord getDatastream(String pid, String dsid) {
200        try {
201            return new RESTFedoraDatastreamRecordImpl(fc, pid, dsid);
202        } catch (FedoraClientException e) {
203            throw new RuntimeException(e);
204        }
205    }
206
207    /**
208     * {@inheritDoc}
209     */
210    public boolean doesDatastreamExist(String pid, String dsid) {
211        try {
212            return FedoraClient.getDatastream(pid, dsid).execute(fc)
213                    .getStatus() == 200;
214        } catch (FedoraClientException ex) {
215            throw new RuntimeException(ex);
216        }
217    }
