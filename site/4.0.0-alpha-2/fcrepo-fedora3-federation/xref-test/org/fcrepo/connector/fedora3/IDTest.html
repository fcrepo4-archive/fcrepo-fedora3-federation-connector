28      @Test
29      public void testRootId() {
30          Assert.assertTrue("The static root id should be the root id.",
31                  ID.ROOT_ID.isRootID());
32          Assert.assertFalse("The root id should not be an object id.",
33                  ID.ROOT_ID.isObjectID());
34          Assert.assertFalse("The root id should not be a datastream id.",
35                  ID.ROOT_ID.isDatastreamID());
36          Assert.assertTrue("The root id should be \"/\".", ID.ROOT_ID.getId().equals("/"));
37          Assert.assertTrue("The item with id \"/\" should be root.", new ID("/").isRootID());
38          Assert.assertNull("The root id should have no parent.", ID.ROOT_ID.getParentId());
39          Assert.assertTrue("The root name should have at least one character.", ID.ROOT_ID.getName().length() >= 1);
40          Assert.assertNull("The root id should not be associated with any pid.", ID.ROOT_ID.getPid());
41          Assert.assertEquals("The id should be the same as the web path.", ID.ROOT_ID.getId(), ID.ROOT_ID.getURLPath());
42      }

44      @Test
45      public void testObjectId() {
46          String pid = "changeme:1";
47          ID id = ID.objectID(pid);
48          Assert.assertTrue("The object ID should  be an object id.",
49                  id.isObjectID());
50          Assert.assertFalse("The object ID should not be a root id.",
51                  id.isRootID());
52          Assert.assertFalse("The object ID should not be a datastream id.",
53                  id.isDatastreamID());
54          Assert.assertTrue("The object ID should retain the pid.",
55                  id.getPid().equals(pid));
56          Assert.assertTrue("The object ID be the parent of the root id.",
57                  id.getParentId().equals(ID.ROOT_ID.getId()));
58          Assert.assertNull("The object ID should not contain a dsid.",
59                  id.getDSID());
60          Assert.assertEquals("The id should be the same as the web path.",
61                  id.getId(), id.getURLPath());
62      }

64      @Test
65      public void testDatastreamId() {
66          String pid = "changeme:1";
67          String dsId = "RELS-EXT";
68          ID objectId = ID.objectID(pid);
69          ID id = ID.datastreamID(pid, dsId);
70          Assert.assertTrue("The datastream ID should  be a datastream id.",
71                  id.isDatastreamID());
72          Assert.assertFalse("The datastream ID should not be a root id.",
73                  id.isRootID());
74          Assert.assertFalse("The datastream ID should not be an object id.",
75                  id.isObjectID());
76          Assert.assertTrue("The datastream ID should retain the pid.",
77                  id.getPid().equals(pid));
78          Assert.assertEquals("The datastream ID be the parent of an object id.",
79                  id.getParentId(), objectId.getId());
80          Assert.assertEquals("The datastream ID should contain a dsid.",
81                  id.getDSID(), dsId);
82          Assert.assertEquals("The id should be the same as the web path.",
83                  id.getId(), id.getURLPath());
84      }

86      @Test
87      public void testContentId() {
88          String pid = "changeme:1";
89          String dsId = "RELS-EXT";
90          ID id = ID.contentID(pid, dsId);
91          ID datastreamId = ID.datastreamID(pid, dsId);
92          Assert.assertTrue("The content ID should  be a content id.",
93                  id.isContentID());
94          Assert.assertFalse("The content ID should not be a root id.",
95                  id.isRootID());
96          Assert.assertFalse("The content ID should not be an object id.",
97                  id.isObjectID());
98          Assert.assertFalse("The content ID should not be a datastream id",
99                  id.isDatastreamID());
100         Assert.assertTrue("The content ID should retain the pid.",
101                 id.getPid().equals(pid));
102         Assert.assertEquals("The content ID be the child of a datastream id.",
103                 id.getParentId(), datastreamId.getId());
104         Assert.assertEquals("The content ID should contain a dsid.",
105                 id.getDSID(), dsId);
106         Assert.assertEquals("The content ID should be the JCR constant.",
107                 id.getName(), JcrConstants.JCR_CONTENT);
108         Assert.assertEquals("The id should not be the same as the web path.",
109                 id.getId(),
110                 id.getURLPath().replace("fcr:content", "jcr:content"));
111     }

113         /**
114          * The pattern for names vs ids is such that an item's id is that
115          * item's parent's id followed by a '/' and then the items name.
116          */
117         @Test
118         public void testNamePattern() {
119             ID contentId = ID.contentID("changeme:1", "RELS-EXT");
120             Assert.assertEquals("An item's id should be the concatenation of its parent's id, a slash and it's name.", contentId.getId(), contentId.getParentId() + "/" + contentId.getName());
121             ID dsId = new ID(contentId.getParentId());
122             Assert.assertEquals("An item's id should be the concatenation of its parent's id, a slash and it's name.", dsId.getId(), dsId.getParentId() + "/" + dsId.getName());
123         }
