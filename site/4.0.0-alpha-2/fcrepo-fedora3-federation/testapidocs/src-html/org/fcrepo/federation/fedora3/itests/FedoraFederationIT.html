162    @Test
163    public void testDatastreamContent() throws Exception {
164        String url = serverAddress + "rest/f3/" + ID.contentID(pid, dsid).getURLPath();
165        HttpGet testContent = new HttpGet(url);
166        assertEquals("Fetching datastream content at " + url, 200, getStatus(testContent));
167        final HttpResponse response = client.execute(testContent);
168        logger.debug("Returned from HTTP GET, now checking content...");
169        assertEquals("Content must be identical to that in fedora 3!", "simple text\n",
170                EntityUtils.toString(response.getEntity()));
171    }

173    public static void ingestFoxml(String pid) throws FedoraClientException {
174        FedoraClient.ingest(pid).content(FedoraFederationIT.class.getClassLoader().getResourceAsStream("foxml/" + pid.replace(':', '_') + ".xml")).execute(fc);
175    }
176
177    protected HttpResponse execute(final HttpUriRequest method)
178            throws ClientProtocolException, IOException {
179        logger.debug("Executing: " + method.getMethod() + " to " +
180                method.getURI());
181        return client.execute(method);
182    }
183
184    protected int getStatus(final HttpUriRequest method)
185            throws ClientProtocolException, IOException {
186        HttpResponse response = execute(method);
187        int result = response.getStatusLine().getStatusCode();
188        if (!(result > 199) || !(result < 400)) {
189            logger.warn(EntityUtils.toString(response.getEntity()));
190        }
191        return result;
192    }
