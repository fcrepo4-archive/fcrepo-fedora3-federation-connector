017package org.fcrepo.connector.fedora3.rest;
018
019import com.yourmediashelf.fedora.client.FedoraClientException;
020import com.yourmediashelf.fedora.generated.access.ObjectDatastreams;
021import com.yourmediashelf.fedora.generated.access.ObjectProfile;
022import org.fcrepo.connector.fedora3.FedoraObjectRecord;
023import org.junit.Assert;
024import org.junit.Test;
025
026import javax.xml.bind.JAXBContext;
027import javax.xml.bind.JAXBException;
028import javax.xml.bind.Unmarshaller;
029import java.text.ParseException;
030import java.util.Arrays;
031import java.util.Collections;
032import java.util.Date;
033
034/**
035 * @Michael Durbin
036 */
037public class ObjectProfileObjectRecordImplTest {
038
039    public static ObjectProfile parseObjectProfile(String resourceName) throws JAXBException {
040        JAXBContext context = JAXBContext.newInstance(ObjectProfile.class);
041        Unmarshaller unmarshaller = context.createUnmarshaller();
042        return (ObjectProfile) unmarshaller.unmarshal(ObjectProfileObjectRecordImplTest.class.getClassLoader().getResource(resourceName));
043    }
044
045    public static ObjectDatastreams parseListDatastreamsResponse(String resourceName) throws JAXBException {
046        JAXBContext context = JAXBContext.newInstance(ObjectDatastreams.class);
047        Unmarshaller unmarshaller = context.createUnmarshaller();
048        return (ObjectDatastreams) unmarshaller.unmarshal(ObjectProfileObjectRecordImplTest.class.getClassLoader().getResource(resourceName));
049    }
050
051    @Test
052    public void testObjectProfile() throws JAXBException, FedoraClientException, ParseException {
053        FedoraObjectRecord rec = new ObjectProfileObjectRecordImpl(parseObjectProfile("mocked-responses/test_1/object-profile.xml"), parseListDatastreamsResponse("mocked-responses/test_1/list-datastreams.xml").getDatastream());
054        Assert.assertEquals("PID should be parsed properly.", "test:1", rec.getPid());
055        Assert.assertEquals("State should be parsed properly.", "A", rec.getState());
056        Assert.assertEquals("Label should be parsed properly.", "page 1", rec.getLabel());
057        Assert.assertEquals("OwnerID should be parsed properly.", Collections.singletonList("fedoraAdmin"), rec.getOwnerIds());
058        Assert.assertEquals("Modification date should be parsed properly.", new Date(1368637379464L), rec.getModificationDate());
059        Assert.assertEquals("Creation date should be parsed properly.", new Date(1304029580911L), rec.getCreatedDate());
060        Assert.assertEquals("Datastream id list should be parsed properly.", Arrays.asList(new String[]{"DC", "RELS-EXT"}), rec.listDatastreamIds());
061    }
