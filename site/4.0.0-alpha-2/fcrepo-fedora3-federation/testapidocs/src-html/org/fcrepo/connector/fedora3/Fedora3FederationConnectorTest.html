043public class Fedora3FederationConnectorTest implements FedoraJcrTypes {

045    @Mock Fedora3DataInterface mockF3;
046
047    @Mock BinaryValue mockBinaryValue;

052
053    Fedora3FederationConnector c;

055    /**
056     * Sets up a Fedora3DataInterface that pretends to be a fedora repository
057     * that contains two objects, "changeme:1" and "changeme:2".  The object
058     * "changeme:2" has two datastreams: "DC" and "RELS-EXT".
059     */
060    @Before
061    public void setUp() throws Exception {
062        MockitoAnnotations.initMocks(this);
063
064        when(mockF3.getObjectPids(anyInt(), anyInt())).thenReturn(Arrays.asList(new String[] { "changeme:1", "changeme:2" }));
065        FedoraObjectRecord changeme1 = new MockObjectRecord("changeme:1");
066        when(mockF3.getObjectByPid("changeme:1")).thenReturn(changeme1);
067        FedoraObjectRecord changeme2
068            = new MockObjectRecord("changeme:2", "DC", "RELS-EXT");
069        when(mockF3.getObjectByPid("changeme:2")).thenReturn(changeme2);
070        when(mockF3.getDatastream("changeme:2", "DC")).thenReturn(
071                RESTFedoraDatastreamRecordImplTest.getDSRecord(
072                "mocked-responses/changeme_2/dc-datastream-history.xml"));
073        when(mockF3.doesObjectExist("changeme:1")).thenReturn(true);
074        when(mockF3.doesObjectExist("changeme:2")).thenReturn(true);
075        when(mockF3.doesDatastreamExist("changeme:2", "DC")).thenReturn(true);
076        when(mockF3.doesDatastreamExist("changeme:2", "RELS-EXT")).thenReturn(true);
077
078        c = new MockedFedora3FederationConnector();
079        c.f3 = mockF3;
080        c.organizer = new FlatTruncatedOrganizer(mockF3);
081    }
082
083    @Test
084    public void testInitialization() throws Exception {
085        try {
086            new Fedora3FederationConnector().initialize(null, null);
087            Assert.fail("Initialization should fail without having set required properties.");
088        } catch (RepositoryException e) {
089        }
090
091        Fedora3FederationConnector c = new Fedora3FederationConnector();
092        c.fedoraUrl = "malformed-url";
093        c.username = "username";
094        c.password = "password";
095        try {
096            c.initialize(null, null);
097        } catch (RepositoryException e) {
098            Assert.assertEquals("Initialization should fail because of bad URL.", e.getCause().getClass(), java.net.MalformedURLException.class);
099        }
100    }
101
102    @Test
103    public void testGetDocumentById() throws Exception {
104        Assert.assertNotNull("The root object is exposed by the federation.", c.getDocumentById(ID.ROOT_ID.getId()));
105        Assert.assertNotNull("The object \"changeme:1\" is exposed by the federation.", c.getDocumentById(ID.objectID("changeme:1").getId()));
106        Assert.assertNotNull("The object \"changeme:2\" is exposed by the federation.", c.getDocumentById(ID.objectID("changeme:2").getId()));
107        Assert.assertNotNull("The datastream \"DC\" on \"changeme:2\" exists.", c.getDocumentById(ID.datastreamID("changeme:2", "DC").getId()));
108        Assert.assertNotNull("The content of datastream \"DC\" on \"changeme:2\" exists.", c.getDocumentById(ID.contentID("changeme:2", "DC").getId()));
109
110    }
111
112    @Test
113    public void testHasDocument() {
114        Assert.assertTrue("Root document should exist.", c.hasDocument(ID.ROOT_ID.getId()));
115        Assert.assertTrue("Document for \"changeme:1\" should exist.", c.hasDocument(ID.objectID("changeme:1").getId()));
116        Assert.assertTrue("Document for \"DC\" datastream on \"changeme:2\" should exist.", c.hasDocument(ID.datastreamID("changeme:2", "DC").getId()));
117        Assert.assertTrue("Document for content of \"DC\" datastream on \"changeme:2\" should exist.", c.hasDocument(ID.contentID("changeme:2", "DC").getId()));
118    }
119

134    }

136    /**
137     * Overrides certain methods to allow for unit testing.
138     */
139    private class MockedFedora3FederationConnector extends Fedora3FederationConnector {
140
141        /**
142         * Overrides the implementation in org.modeshape.jcr.federation.spi.Connector to
143         * return a dummy implementation.
144         */
145        public DocumentWriter newDocument(String id) {
146            when(mockumentWriter.document()).thenReturn(mockument);
147            return mockumentWriter;
148        }
149    }

151    /**
152     * A mock FedoraObjectRecord instance with minimal parameters.
153     */
154    private class MockObjectRecord extends AbstractFedoraObjectRecord {
155
156        protected MockObjectRecord(String pid, String ... dsids) {
157            super(pid, new Date(), new Date());
158            super.datastreams = Arrays.asList(dsids);
159        }
160    }
