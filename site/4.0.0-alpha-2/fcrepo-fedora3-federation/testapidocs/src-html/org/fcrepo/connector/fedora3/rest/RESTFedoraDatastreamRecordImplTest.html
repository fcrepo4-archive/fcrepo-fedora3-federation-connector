017package org.fcrepo.connector.fedora3.rest;
018
019import com.yourmediashelf.fedora.generated.management.DatastreamHistory;
020import org.fcrepo.connector.fedora3.FedoraDatastreamVersionRecord;
021import org.junit.Assert;
022import org.junit.Test;
023
024import javax.xml.bind.JAXBContext;
025import javax.xml.bind.JAXBException;
026import javax.xml.bind.Unmarshaller;
027import java.util.Arrays;
028import java.util.Date;
029
030/**
031 * Because the RESTFedoraDatastreamRecordImpl class is a quick and dirty
032 * implementation that is intended to be a proof of concept, and because
033 * future integration tests will fully cover the code, these unit tests
034 * only cover the small amount of code that represents logic that could
035 * be reused.
036 *
037 * @author Michael Durbin
038 */
039public class RESTFedoraDatastreamRecordImplTest {
040
041    public static DatastreamHistory parseDatastreamHistory(String resourceName) throws JAXBException {
042        JAXBContext context = JAXBContext.newInstance(DatastreamHistory.class);
043        Unmarshaller unmarshaller = context.createUnmarshaller();
044        return (DatastreamHistory) unmarshaller.unmarshal(RESTFedoraDatastreamRecordImplTest.class.getClassLoader().getResource(resourceName));
045    }
046
047    public static RESTFedoraDatastreamRecordImpl getDSRecord(String resourceName) throws JAXBException {
048        RESTFedoraDatastreamRecordImpl ds = new RESTFedoraDatastreamRecordImpl(null, parseDatastreamHistory(resourceName).getDatastreamProfile());
049        return ds;
050    }
051
052    @Test
053    public void testMockedRecord() throws Exception {
054        RESTFedoraDatastreamRecordImpl testRecord = getDSRecord("mocked-responses/mock_1/dc-datastream-history.xml");
055        Assert.assertEquals("Pid is preserved.", "mock:1", testRecord.getPid());
056        Assert.assertEquals("DSID is preserved.", "DC", testRecord.getId());
057        Assert.assertEquals("Control group is preserved.", "X",
058                testRecord.getControlGroup());
059        Assert.assertEquals("State is preserved.", "A", testRecord.getState());
060        Assert.assertTrue("Versionable is preserved.", testRecord.getVersionable());
061
062        FedoraDatastreamVersionRecord c = testRecord.getCurrentVersion();
063        Assert.assertEquals("Label is preserved.", "Dublin Core Record for this object", c.getLabel());
064        Assert.assertEquals("Version ID is preserved.", "DC.3", c.getVersionId());
065        Assert.assertEquals("Created Date is preserved.", new Date(1380634413335L), c.getCreatedDate());
066        Assert.assertEquals("Mime type is preserved.", "text/xml", c.getMimeType());
067        Assert.assertEquals("Format URI is preserved.", "http://www.openarchives.org/OAI/2.0/oai_dc/", c.getFormatURI());
068        Assert.assertEquals("No alt ids are present.", Arrays.asList(new String[] { "oai_dc", "dublin_core" }), c.getAltIDs());
069        Assert.assertEquals("Content digest type is preserved.", "SHA-1", c.getContentDigestType());
070        Assert.assertEquals("Content digest is preserved.", "920a51eb1824168e986bbb0ad7d14f81db28d8e1", c.getContentDigest());
071        Assert.assertEquals("Content length is preserved.", 423, c.getContentLength());
072        Assert.assertArrayEquals("SHA-1 needs not be computed.", RESTFedoraDatastreamRecordImpl.getSha1BytesFromHexString(c.getContentDigest()), c.getSha1());
073        try {
074            FedoraDatastreamVersionRecord p = testRecord.getHistory().get(3);
075            Assert.assertNull("Content digest does not exist on first version.", p.getContentDigest());
076            p.getSha1(); // should throw exception with this mock object
077            Assert.fail("SHA-1 cannot be computed on this mock object.");
078        } catch (NullPointerException ex) {
079        }
080    }
081
082    @Test
083    public void testGetSha1() {
084        byte[] bytes = new byte[] { 0x01, 0x23, 0x45, 0x67, (byte) 0x89, (byte) 0xAB, (byte) 0xCD, (byte) 0xEF };
085        String string = "0123456789ABCDEF";
086        Assert.assertTrue(Arrays.equals(RESTFedoraDatastreamRecordImpl.getSha1BytesFromHexString(string), bytes));
087    }
088}
