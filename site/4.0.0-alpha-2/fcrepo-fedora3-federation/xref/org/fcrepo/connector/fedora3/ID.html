17  package org.fcrepo.connector.fedora3;
18  
19  import org.fcrepo.jcr.FedoraJcrTypes;
20  import org.modeshape.common.text.Jsr283Encoder;
21  import org.modeshape.jcr.api.JcrConstants;
22  
23  /**
24   * Encapsualtes the logic associated with mapping ids within the
25   * {@link Fedora3FederationConnector} to the types of objects each represents.
26   * 
27   * Objects within the federation can fall into 4 types:
28   * <ul>
29   *   <li>root - the ID for the federation node itself</li>
30   *   <li>
31   *     object - the ID of a node representing an object in the fedora 3
32   *     repository that is being federated over
33   *   </li>
34   *   <li>
35   *     datastream - the ID of a node representing a datastream from an object
36   *     in the fedora 3 repository that is being federated over
37   *   </li>
38   *   <li>content - the ID of content from a datastream</li>
39   * </ul>
40   * <p>
41   *   The IDs are meant to be opaque and the implementation may change in later
42   *   versions of this class, but for reference, the current implementation
43   *   creates ids in the following pattern. /, /pid, /pid/dsid, /pid/dsid/content
44   * </p>
45   * 
46   * @author Michael Durbin
47   */
48  public class ID implements FedoraJcrTypes {
49  
50      private String id;
51  
52      /**
53       * A constructor that accepts a known id.
54       */
55      public ID(String id) {
56          this.id = id;
57      }
58  
59      /**
60       * Gets a name for the node with this id.
61       */
62      public String getName() {
63          if (isRootID()) {
64              return "/";
65          } else if (isContentID()) {
66              return JcrConstants.JCR_CONTENT;
67          } else {
68              return id.substring(id.lastIndexOf('/') + 1);
69          }
70      }
71  
72      /**
73       * Gets the ID within the federation.
74       */
75      public String getId() {
76          return id;
77      }
78  
79      /**
80       * Gets the URL path (relative to the federation root) that would address
81       * the node with this id.
82       */
83      public String getURLPath() {
84          if (isContentID()) {
85              return getId().replace(JcrConstants.JCR_CONTENT, FCR_CONTENT);
86          } else {
87              return getId();
88          }
89      }
90  
91      /**
92       * Gets the id for the parent within the federation of the node with
93       * this id.
94       */
95      public String getParentId() {
96          if (isRootID()) {
97              return null;
98          } else if (isObjectID()) {
99              return ROOT_ID.id;
100         } else if (isDatastreamID()) {
101             return objectID(getPid()).id;
102         } else {
103             assert(isContentID());
104             return datastreamID(getPid(), getDSID()).id;
105         }
106     }
107 
108     /**
109      * Determines if the id is the root id.
110      */
111     public boolean isRootID() {
112         return "/".equals(id);
113     }
114 
115     /**
116      * Determines if the id is for a fedora 3 object.
117      */
118     public boolean isObjectID() {
119         return id.split("/").length == 1;
120     }
121 
122     /**
123      * Gets the PID associated with this id.  For "object" IDs this will be the
124      * PID of the object, for "datastream" IDs this will be the pid to which
125      * the datastream belongs and for "content" IDs this will be the PID of the
126      * object that contains the datastream whose content node has the given id.
127      * @return the pid, if one can be determined from the id.  This should
128      * return a non-null value unless this id is the root id.
129      */
130     public String getPid() {
131         if (isRootID()) {
132             return null;
133         } else {
134             String[] path = id.split("/");
135             return new Jsr283Encoder().decode(path[0]);
136         }
137     }
138 
139     /**
140      * Gets the DSID associated with this id or null if the object represented
141      * by the node with this id does not pertain to a datastream or its
142      * content.
143      */
144     public String getDSID() {
145         String[] path = id.split("/");
146         if (path.length < 2) {
147             return null;
148         } else {
149             return new Jsr283Encoder().decode(path[1]);
150         }
151     }
152 
153     /**
154      * Determines if the id is for a fedora 3 datastream.
155      */
156     public boolean isDatastreamID() {
157         return id.split("/").length == 2;
158     }
159 
160     /**
161      * Determines if the id is for a fedora 3 datastream content node.
162      */
163     public boolean isContentID() {
164         return id.split("/").length == 3;
165     }
166 
167     /**
168      * The ID for the root of the federation.
169      */
170     public static final ID ROOT_ID = new ID("/");
171 
172     /**
173      * Gets the ID for the node within the federation representing an object
174      * with the given pid.
175      */
176     public static ID objectID(String pid) {
177         return new ID(new Jsr283Encoder().encode(pid));
178     }
179 
180     /**
181      * Gets the ID for the node within the federation representing a datastream
182      * with the given dsid on the object in the federation having the given
183      * pid.
184      */
185     public static ID datastreamID(String pid, String datastream) {
186         return new ID(new Jsr283Encoder().encode(pid) + "/"
187                 + new Jsr283Encoder().encode(datastream));
188     }
189 
190     /**
191      * Gets the ID for the node within the federation representing datastream
192      * content with the given dsid on the object in the federation having the
193      * given pid.
194      */
195     public static ID contentID(String pid, String datastream) {
196         return new ID(new Jsr283Encoder().encode(pid) + "/"
197                 + new Jsr283Encoder().encode(datastream) + "/"
198                 + JcrConstants.JCR_CONTENT);
199     }
200 }
